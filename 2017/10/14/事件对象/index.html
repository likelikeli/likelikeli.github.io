事件对象

键盘事件对象

keydown支持特殊按键，比如ALT CTRL Command等等，但不区分大小写A和a的keyCode都是65
keypress不支持特殊按键，但区分大小写

document.onkeydown=function(){
    // 获取事件对象可以使用window.event。 注意：在火狐获取不到
    console.log(window.event);
    // 在火狐需要通过参数获取
    //console.log(ev);
    // 兼容性的写法
    var evObj=window.event || ev ;
    console.log(evObj.keyCode);
    // 如何判断是否按了组合键
    if(evObj.keyCode==67 && evObj.metaKey){
        console.log("按了commond+c")
    }

一个简单的键盘事件小例子，利用方向键使div移动

<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8">
        <title>使用方向键使div移动</title>
        <style type="text/css">
            body{
                margin: 0 auto;
                position: relative;
            }
            .redDiv{
                width: 100px;
                height: 100px;
                background: red;
                position: absolute;
            }
        </style>
    </head>
    <body>
        <div class="redDiv"></div>
    </body>
    <script type="text/javascript">
        var redDiv=document.getElementsByClassName("redDiv")[0];
        var boolArr=[false,false,false,false];
        document.onkeydown=function(ev){
            var evObj=window.event || ev ;
            boolArr[evObj.keyCode-37]=true;
        }
        document.onkeyup=function(ev){
            var evObj=window.event || ev ;
            boolArr[evObj.keyCode-37]=false;
        }
        setInterval(function(){
            var speed=5;
            var leftSpeed=(boolArr[2]-boolArr[0])*speed;
            var topSpeed=(boolArr[3]-boolArr[1])*speed;
            redDiv.style.left=redDiv.offsetLeft+leftSpeed+"px";
            redDiv.style.top=redDiv.offsetTop+topSpeed+"px";
        },20)
    </script>
</html>

鼠标事件对象

var box=document.getElementsByClassName("box")[0];
var redDiv=document.getElementsByClassName("redDiv")[0];
box.onclick=function(ev){
    var evObj=window.event || ev;
    // evObj.target触发事件的元素，不一定是绑定事件的元素
    //console.log(evObj.target);
    console.log(evObj.offsetY);
    //触发事件的元素，不是绑定的元素，鼠标距离自身的Y轴距离
    console.log(evObj.clientY);
    //鼠标距离屏幕的Y轴的距离
    console.log(evObj.pageY);
    //鼠标距离页面的Y轴的距离
}

鼠标拖拽使元素移动

<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8">
        <title>鼠标拖拽</title>
        <style type="text/css">
            #redDiv{
                width: 100px;
                height: 100px;
                background: red;
                position: absolute;
            }
            body{
                height: 2000px;
                position: relative;
            }
            #box{
                width: 600px;
                height: 600px;
                border: 10px solid green;
                position: absolute;
                top: 1000px;
                left: 200px;
            }
        </style>
    </head>
    <body>
        <div id="box">
            <div id="redDiv"></div>
        </div>        
    </body>
    <script type="text/javascript">
        var redDiv=document.getElementById("redDiv");
        var box=document.getElementById("box");
        redDiv.onmousedown=function(ev){
            var offsetX=ev.offsetX;
            var offsetY=ev.offsetY;
            document.onmousemove=function(ev){
                var evObj=window.event || ev ;
                var x=ev.pageX-box.offsetLeft-offsetX-18;
                var y=ev.pageY-box.offsetTop-offsetY-18;                
                if(x<0){
                    x=0;
                }else if(x>500){
                    x=500;
                }
                if(y<0){
                    y=0;
                }else if(y>500){
                    y=500;
                }
                redDiv.style.left=x+"px";
                redDiv.style.top=y+"px";
            }
        }
        redDiv.onmouseup=function(){
            document.onmousemove=null;
        }
    </script>
</html>
