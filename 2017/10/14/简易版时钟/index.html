简易版时钟

自己编写的简单的时钟，代码很简单，很容易看懂
代码：

<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8">
        <title>时钟</title>
        <style type="text/css">
            .wrap{
                width: 300px;
                height: 300px;
                border: 10px solid black;
                box-shadow: 2px 0px 3px 3px gray inset;
                border-radius: 50%;
                position: relative;
                margin: 200px auto;
                background: url("1.jpg")no-repeat;
                background-size: 300px 300px;
            }
            .box{
                width: 280px;
                height: 20px;
                position: absolute;
                left: 0;
                top: 0;
                right: 0;
                bottom: 0;
                margin: auto;                
            }
            .box span{
                width: 20px;
                height: 20px;
                line-height: 20px;
                text-align: center;
                float: left;
                color: white;
            }            
            .box1{
                width: 10px ;
                height: 10px;
                border-radius: 50%;
                background: black;
                position: absolute;
                left: 0;
                top: 0;
                right: 0;
                bottom: 0;
                margin: auto;
                z-index: 4;
            }
            #ho{
                width: 10px;
                height: 80px;
                background: green;
                position: absolute;
                transform-origin:5px 100%;
                top: 70px;
                left: 145px;
                border-top-left-radius: 5px 60px;
                border-top-right-radius: 5px 60px;
                z-index: 3;
            }
            #min{
                width: 8px;
                height: 100px;
                background: red;
                position: absolute;
                transform-origin:4px 100%;
                top: 50px;
                left: 146px;
                border-top-left-radius: 5px 60px;
                border-top-right-radius: 5px 60px;
                z-index: 2;
            }
            #sec{
                width: 6px;
                height: 120px;
                background: blue;
                position: absolute;
                border-top-left-radius: 5px 60px;
                border-top-right-radius: 5px 60px;
                transform-origin:3px 100%;
                top: 30px;
                left: 147px;
            }
        </style>
    </head>
    <body>
        <div class="wrap">
            <!-- div.box(span{$})*12 -->
            <div class="box"><span>9</span></div>
            <div class="box"><span>10</span></div>
            <div class="box"><span>11</span></div>
            <div class="box"><span>12</span></div>
            <div class="box"><span>1</span></div>
            <div class="box"><span>2</span></div>
            <div class="box"><span>3</span></div>
            <div class="box"><span>4</span></div>
            <div class="box"><span>5</span></div>
            <div class="box"><span>6</span></div>
            <div class="box"><span>7</span></div>
            <div class="box"><span>8</span></div>
            <div class="box1"></div>
            <div id="ho"></div>
            <div id="min"></div>
            <div id="sec"></div>
        </div>
    </body>
    <script type="text/javascript">
        var box=document.getElementsByClassName("box");
        var span=document.getElementsByTagName("span");
        var ho=document.getElementById("ho");
        var min=document.getElementById("min");
        var sec=document.getElementById("sec");
        for(var i=0;i<12;i++){
            a=30*i;
            b=(-30)*i;
            box[i].style.transform='rotate('+a+'deg)';
            span[i].style.transform='rotate('+b+'deg)';
        }
        var date=new Date();
            var minutes=date.getMinutes();
            var hours=date.getHours();
            var seconds=date.getSeconds();
            var c=hours*30+(minutes/60)*30+(seconds/60/60)*30;
            var d=minutes*6+(seconds/60)*6;
            var e=seconds*6;
            ho.style.transform='rotate('+c+'deg)';
            min.style.transform='rotate('+d+'deg)';
            sec.style.transform='rotate('+e+'deg)';
        setInterval(function(){
            var date=new Date();
            var minutes=date.getMinutes();
            var hours=date.getHours();
            var seconds=date.getSeconds();
            var c=hours*30+(minutes/60)*30+(seconds/60/60)*30;
            var d=minutes*6+(seconds/60)*6;
            var e=seconds*6;
            ho.style.transform='rotate('+c+'deg)';
            min.style.transform='rotate('+d+'deg)';
            sec.style.transform='rotate('+e+'deg)';
        },1000)        
    </script>
</html>
